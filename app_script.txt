from kivy.app import App
from kivy.uix.label import Label
from kivy.clock import Clock

class MyApp(App):
    def build(self):
        Clock.schedule_once(self.send_notification, 0.5)
        return Label(text="App is running!", font_size="20sp", halign="center")

    def send_notification(self, dt):
        try:
            from plyer import notification
            notification.notify(
                title="Hello",
                message="Hello testing",
                app_name="MyApp",
                timeout=10,
            )
        except Exception as e:
            print(e)

if __name__ == "__main__":
    MyApp().run()
